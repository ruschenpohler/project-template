@"
# Auto detect text files and normalize line endings to LF in the repository
* text=auto

# Force LF for specific file types
*.py text eol=lf
*.sh text eol=lf
*.md text eol=lf
*.yaml text eol=lf
*.yml text eol=lf
*.json text eol=lf

# Windows-specific files should use CRLF
*.bat text eol=crlf
*.ps1 text eol=crlf
"@ | Out-File -FilePath .gitattributes -Encoding utf8
